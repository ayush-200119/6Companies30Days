    long long calc(long long int N,int R)
    {
        if(R==1)
        return (N%mod);
        else
        {
            if(R%2==0){
            long long int temp=(calc(N,R/2)%mod);
            temp=(temp*temp)%mod;
            return temp;
            }
            else
            {
                return ((N%mod)*(calc(N,R-1)%mod))%mod;
            }
        }
    }
    
    long long power(int N,int R)
    {
       //Your code here
       long long res=calc(N,R);
       return (res%mod);
        
    }

};

long long rev(long long n)
{

     while(n > 0)
